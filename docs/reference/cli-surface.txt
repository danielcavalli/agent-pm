- --tech-stack <items...> | description=Technology stack entries (repeatable) (default: []) | default=[]
- --architecture <pattern> | description=Architecture pattern label (default: ) | default=

## pm story add <epicCode>
description: Add a new story to an epic (e.g. pm story add E001 --title "..." or pm story add PM-E001 --title "...")
arguments:
- <epicCode>
options:
- --title <title> | description=Story title (required) | required=yes

- --priority <priority> | description=Story priority: high | medium | low (default: medium) | default=medium
- --criteria <criteria...> | description=Acceptance criteria entries (repeatable) (default: []) | default=[]
- --depends-on <storyCode...> | description=Story dependencies (repeatable) (default: []) | default=[]

## pm adr
description: Manage Architecture Decision Records (ADRs)
arguments:
- none
options:
- none

## pm adr create
description: Create a new ADR (e.g. pm adr create --project PM --title ...)
arguments:
- none
options:
- --project <code> | description=Project code (required) | required=yes
- --title <title> | description=ADR title (required) | required=yes
- --status <status> | description=ADR lifecycle status: proposed | accepted | deprecated | superseded (required) | required=yes
- --context <context> | description=Decision context (required) | required=yes
- --decision <decision> | description=Decision statement (required) | required=yes
- --positive <items...> | description=Positive consequences (repeatable) (default: []) | default=[]
- --negative <items...> | description=Negative consequences (repeatable) (default: []) | default=[]
- --author-type <type> | description=ADR author type: agent | human (default: human) | default=human
- --author <name> | description=Human author name
- --author-id <id> | description=Agent author identifier
- --tags <tags...> | description=ADR tags (repeatable) (default: []) | default=[]

## pm adr list [projectCode]
description: List all ADRs for a project
arguments:
- [projectCode]
options:
- none

## pm adr query [projectCode]
description: Query ADRs with filters, ranked by relevance
arguments:
- [projectCode]
options:
- --id <pattern> | description=ADR id pattern
- --status <status> | description=ADR status filter: proposed | accepted | deprecated | superseded
- --tag <tag> | description=Single-tag filter
- --tags <tags...> | description=Multi-tag filter (repeatable)
- --author-type <type> | description=Author-type filter: agent | human
- --author <author> | description=Author filter
- --search <text> | description=Full-text search string
- --limit <n> | description=Maximum result count (default: 5) | default=5
- --format <fmt> | description=Output format: summary | full (default: summary) | default=summary
- --verbose | description=Show relevance scores in summary mode
- --include-superseded | description=Include superseded and deprecated ADRs
